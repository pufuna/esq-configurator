function updatePowerOptions() { const voltage = document.getElementById('voltage').value; const powerSelect = document.getElementById('power'); powerSelect.innerHTML = ''; Object.keys(modelData[voltage]) .sort((a, b) => a - b) .forEach(power => { const option = document.createElement('option'); option.value = power; option.textContent = `${power} кВт`; powerSelect.appendChild(option); }); updateDependencies(); } function updateDependencies() { const voltage = document.getElementById('voltage').value; const power = document.getElementById('power').value; const data = modelData[voltage][power]; document.getElementById('current').value = data.current; const cellSize = document.getElementById('cellSize'); cellSize.value = data.size; const starter = document.getElementById('starter'); starter.disabled = data.starter; starter.value = data.starter ? 'S' : 'X'; const pulsation = document.getElementById('pulsation'); pulsation.innerHTML = voltage === 'T060' ? '' : ''; updateNomenclature(); } function updateNomenclature() { const params = [ document.getElementById('power').value + 'P', document.getElementById('current').value + 'А', document.getElementById('voltage').value, document.getElementById('material').value, document.getElementById('pulsation').value, document.getElementById('starter').value, document.getElementById('bypass').value, document.getElementById('reactor').value, document.getElementById('ip').value, document.getElementById('expansion').value, document.getElementById('ups').value, document.getElementById('cable').value, document.getElementById('protocol').value, '-' + document.getElementById('cellSize').value, document.getElementById('cellBypass').value ]; document.getElementById('result').textContent = `ESQ F ME800-${params.join('')}`; } document.addEventListener('DOMContentLoaded', () => { updatePowerOptions(); document.querySelectorAll('select').forEach(el => el.addEventListener('change', updateNomenclature) ); });
